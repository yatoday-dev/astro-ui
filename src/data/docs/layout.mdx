# Layout

## Usage

Create `src/layouts/PageLayout.astro` in your project

```astro
---
import { Layout } from 'yt-astro-ui';

// Import your CSS styles
import '~/assets/styles/tailwind.css';

// Use your Favicons and other components if you need to
import Favicons from '~/components/Favicons.astro';
import CustomScripts from '~/components/CustomScripts.astro';
import CustomStyles from '~/components/CustomStyles.astro';

export interface Props {
  metadata?: MetaData;
}

const { metadata } = Astro.props;
---

<Layout metadata={metadata}>
  <Favicons slot="favicons" />
  <CustomScripts slot="scripts" />
  <CustomStyles slot="styles" />

  <!-- Your content -->
</Layout>
```
